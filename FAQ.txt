==JS FAQ==
<script>
    // FAQ Accordion - FIXED & SIMPLE
    document.querySelectorAll('.faq-btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const content = btn.nextElementSibling;
        const isOpen = content.classList.contains('open');

        // close others
        document.querySelectorAll('.faq .content').forEach(c => {
          c.classList.remove('open');
          c.style.maxHeight = null;
          c.style.opacity = 0;
        });
        document.querySelectorAll('.faq-btn span').forEach(s => s.textContent = '+');

        // toggle current
        if(!isOpen){
          content.classList.add('open');
          content.style.maxHeight = content.scrollHeight + 'px';
          content.style.opacity = 1;
          btn.querySelector('span').textContent = '-';
        }
      });
    });

    // Simple form handler (demo)
    function handleForm(e){
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const phone = document.getElementById('phone').value.trim();
      const result = document.getElementById('formResult');
      result.style.display = 'block';
      result.textContent = 'Terima kasih ' + name + '! Kami akan mengirimkan panduan ke WhatsApp ' + phone + ' (simulasi).';
      e.target.reset();
    }
  </script>
